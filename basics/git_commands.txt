sudo apt install git - установка системы контроля версий

git --versions - проверка верии 

git config --global user.email "your@email.com"
git config --global user.name "yourName"
git config --list - проверка насторек 

ssh -keygen - создали ssh ключ

cat ~/.ssh/id_rsa - получение ssh ключа 

создаем репозиторий на github 

переходим в папку с проектом 
git init - превращает текущую директроию в git репозитроий 

git remote add название_связи(обычно -origin)  ссылку на удаленный репозитроий(SSH: git@github:,,,)
git remote -v - просмотр списка удаленных репозиториев
git remote rm навзание связи(обычно - origin) - удаление связи с удаленным репозиторием

git status - проверка статуса файлов

git add название файлов - отслеживание конкретных файлов
git add . -отслеживание всех файлов

git commit - создание коммита(версии кода)
git commit -m 'описание версии' - создание коммита с комментарием 

git push название связи(обычно - origin) 
название ветки - отправка версии кода на удаленный репозиторий   
Например:git push origin master

git clone ссылка на удаленный репозитрий - склонироват/скачать/скопировать удаленные репозиторий

git diff - просмотр внесенных изменений 

git branch - вывод списка веток в терминал (звездочкой помечается активная ветка)

git branch название ветки - создание указаннов ветки

git checkout  название ветки 
                                  - переход на указанную ветку 
git switch  название ветки

git pull название связи название ветки - стягивание изменений с удаленного репозитория с указанной ветки 

git stash - спрятать локальные данные во временную область 
git stash list - просмотр списка спрятанных данных
git stash apply - запись спрятанных данных/извлечение спрятанных данных
git stash drop - удаление спрятанных данных 

git log - просмотр коммитов/версий 
git log --oneline - компактное отображение коммитов
